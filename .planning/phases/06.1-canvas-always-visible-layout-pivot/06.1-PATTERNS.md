**In-place tween pattern (RESEARCH Pattern 4, lines 355-369):**
```typescript
const LIFT_SCALE = 1.3;
const DIM_ALPHA = 0.4;
for (const [cellId, view] of refs.cells) {
  const isLifted = cellId === selectedCellId;
  const targetScale = isLifted ? LIFT_SCALE : 1;
  const targetAlpha = (selectedCellId !== null && !isLifted) ? DIM_ALPHA : 1;
  if (reduceMotion || dt <= 0) {
    view.hex.scale.set(targetScale);
    view.hex.alpha = targetAlpha;
  } else {

    view.hex.alpha = tweenScalar(view.hex.alpha, targetAlpha, dt));

}
```

**`tweenScalar` to reuse** (`src/render/flowgrid/motion.ts:88-90`):
```typescript
export function tweenScalar(current: number, target: number, dt: number): number {
  const k = 1 - Math.exp((-8 * dt) / 1000);
  return current + (target - current) * k;
}
```
